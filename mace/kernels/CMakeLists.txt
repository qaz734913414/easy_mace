set(module_dir "kernels")

# module files dir
set(EASY_MACE_HEDAER_DIR "mace/${module_dir}")

file(GLOB_RECURSE EASY_MACE_MODULE_ALL_HEADERS "${PROJECT_SOURCE_DIR}/${EASY_MACE_SOURCE_DIR}/*.h")

if(USE_OPENCL)
  file(GLOB_RECURSE EASY_MACE_MODULE_ALL_SOURCES "${PROJECT_SOURCE_DIR}/${EASY_MACE_SOURCE_DIR}/*.cc")
  file(GLOB_RECURSE EASY_MACE_MODULE_CL_SOURCES "${PROJECT_SOURCE_DIR}/${EASY_MACE_SOURCE_DIR}/*.cl")
  list(APPEND EASY_MACE_MODULE_ALL_SOURCES ${EASY_MACE_MODULE_CL_SOURCES})
else()
  file(GLOB_RECURSE EASY_MACE_MODULE_ALL_SOURCES "${PROJECT_SOURCE_DIR}/${EASY_MACE_SOURCE_DIR}/*.cc")
endif()

list(APPEND EASY_MACE_ALL_HEADERS ${EASY_MACE_MODULE_ALL_HEADERS})
